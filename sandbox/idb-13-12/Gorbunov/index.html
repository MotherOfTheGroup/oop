Тег <div>

<div> изначально предназначен для выделения фрагмента документа с целью изменения вида содержимого с помощью стилей. В настоящее время это ключевой тег, использующийся для HTML-разметки. Это блочный элемент, которым делят станицу на разделы и группируют блоки других элементов.

Для наглядности кнопка внизу покажет границы всех div на странице.

Код, используемый для формирования списка учащихся на этой странице, подгружается с помощью AJAX оттуда же:

$("#idb-13-12").click(function(){
	$.get({url: "students.json", cache: "false", dataType: "json", success: function( data ) {
		if ($("#idb-13-12-table").length){
			$("#idb-13-12-table").slideUp();
			$("#idb-13-12-table").remove();
		}
		else{
			$("#group-13-12").append("<table class=\"group-table\" id=\"idb-13-12-table\" style=\"display: none\"><tr><th>Имя</th><th>М1</th><th>М2</th></tr></table>");
			$.each(data, function(index, student){
				$("#idb-13-12-table").append("<tr><td><a href=\""+student.href+"\">"+student.name+"</a></td><td>"+student.m1+"</td><td>"+student.m2+"</td></tr>");
			});
			$("#idb-13-12-table").slideDown();
		}
	}});
});
